//@version=5
strategy("BTCUSDT RSI Dip & Spike — 1h Bybit (v2 tuned)", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=100, initial_capital=10000)

// Inputs — tuned via optimizer sweep (best Sortino, 4yr BTC 1h)
rsiLength   = input.int(7,   "RSI Length",       minval=2)
oversold    = input.int(35,  "Oversold Level",   minval=1,  maxval=49)
overbought  = input.int(80,  "Overbought Level", minval=51, maxval=99)

// RSI
rsi = ta.rsi(close, rsiLength)

// Signals
longEntry  = ta.crossunder(rsi, oversold)
longExit   = ta.crossover(rsi,  overbought)

// Orders
if longEntry
    strategy.entry("Long", strategy.long)

if longExit
    strategy.close("Long")

// Plots
hline(overbought, "Overbought", color=color.red,   linestyle=hline.style_dashed)
hline(oversold,   "Oversold",   color=color.green, linestyle=hline.style_dashed)
plot(rsi, "RSI", color=color.blue)
